# [BOJ] 특별한 최단 경로

Problem URL : [특별한 최단 경로](https://www.acmicpc.net/problem/1504)

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <queue>
#define p pair<int,int>
#define MAX_INTEGER 987654321

using namespace std;

vector<p> arr[801];
int N, E;

vector<int> dijk(int start) {
	vector<int> dist(N + 1, MAX_INTEGER);
	dist[start] = 0;
	priority_queue<p> q;
	q.push({ start, 0 });

	while (!q.empty()) {
		int node = q.top().first;
		int cost = q.top().second;
		q.pop();
		if (dist[node] < cost) continue;
		for (int i = 0; i < arr[node].size(); i++) {
			int nextNode = arr[node][i].first;
			int nextCost = arr[node][i].second;
			if (dist[nextNode] > dist[node] + nextCost) {
				dist[nextNode] = dist[node] + nextCost;
				q.push({nextNode, dist[nextNode]});
			}
		}
	}
	return dist;
}

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	cin >> N >> E;
	for (int i = 0; i < E; i++) {
		int a, b, c;
		cin >> a >> b >> c;
		arr[a].push_back({ b, c });
		arr[b].push_back({ a, c });
	}

	int node1, node2, ans1 = 0, ans2 = 0;
	cin >> node1 >> node2;

	vector<int> d;
	d = dijk(1);
	ans1 += d[node1];
	ans2 += d[node2];

	d = dijk(node1);
	ans1 += d[node2];
	ans2 += d[N];

	d = dijk(node2);
	ans1 += d[N];
	ans2 += d[node1];

	int res = min(ans1, ans2);
	if (res >= MAX_INTEGER || res < 0) cout << -1;
	else cout << res;

	return 0;
}
```
