# [BOJ] 히스토그램

Problem URL : [히스토그램](https://www.acmicpc.net/problem/1725)

```cpp
#include<iostream>
#include<stack>
using namespace std;

int main() {
    int n, value;
    scanf("%d", &n);

    stack<pair<int, int>> s;
    long long maxArea = 0;
    long long area;
    int tmpX;

    for (int i = 1; i <= n; i++) {
        scanf("%d", &value);
        tmpX = i;
        while (!s.empty() && s.top().second >= value) {
            // s.top().second 로 만들 수 있는 최대 사각형 넓이
            area = (i - s.top().first) * s.top().second;
            maxArea = maxArea < area ? area : maxArea;
            tmpX = s.top().first;
            s.pop();
        }
        s.push({tmpX, value});
    }

    while (!s.empty()) {
        //남은 stack의 원소들이 오름차순이기에
        //그래프 끝까지가 사각형 밑변의 길이가 된다.
        area = (n + 1 - s.top().first) * s.top().second;
        maxArea = maxArea < area ? area : maxArea;
        s.pop();
    }
    printf("%lld\n", maxArea);
}
```
