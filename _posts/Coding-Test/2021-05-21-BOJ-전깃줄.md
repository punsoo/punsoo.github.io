# [BOJ] 전깃줄

Problem URL : [전깃줄](https://www.acmicpc.net/problem/2565)

```cpp
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int n;
vector<pair<int, int>> pole;
// dp[i]는 수열의 i번째 원소를 마지막으로 하는 부분수열중 최장 길이
int dp[100]; 


int main()
{   
    ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
    cin >> n;
    int a, b;
    for (int i = 0; i < n; i++)
    {
        cin >> a >> b;
        pole.push_back({ a,b });
    }
    sort(pole.begin(), pole.end());

    dp[0] = 1;
    for (int i = 1; i < n; i++)
    {
        dp[i] = 1;
        for (int j = 0; j < i; j++)
        {
            if (pole[i].second > pole[j].second && dp[i] < dp[j] + 1)
            {
                dp[i] = dp[j] + 1;
            }
        }
    }

    int lis = 1;
    for (int i = 1; i < n; i++)
    {
        if (lis < dp[i]) {
            lis = dp[i];
        }
    }

    cout << n - lis << '\n';
}
```

## Comments  
LIS(Longest Increasing Subsequence) 최장 증가 부분 수열 문제이다.
